int binaryGap(int n) {
    int maxGap = 0; 
    int prev = -1;   
    int current = 0; 
    
    while (n > 0) {
        if (n & 1) {  
            if (prev != -1) {  
                maxGap = fmax(maxGap, current - prev);
            }
            prev = current;  
        }
        n >>= 1;  
        current++;  
    }
    
    return maxGap;
}
