bool hasAlternatingBits(int n) {
    if(n == 0) return false;

    bool flag = n & 1;
    n = n >> 1;

    while(n != 0) {
        if((n & 1) == flag) {
            return false;
        }
        flag = !flag;
        n = n >> 1;
    }
    return true;

    /* long int a = n ^ (n >> 1);
    return (a & (a + 1)) == 0;*/
}
