/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
 #include<math.h>
int* countBits(int n, int* returnSize) {
    int* arr = (int*)malloc((n + 1) * sizeof(int));
    if (!arr) return NULL;

    arr[0] = 0;
    *returnSize = n + 1;

    for (int i = 1; i <= n; i++) {
        arr[i] = arr[i >> 1] + (i & 1);
    }

    return arr;
}
