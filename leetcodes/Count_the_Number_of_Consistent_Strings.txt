#include <stdbool.h>

int countConsistentStrings(char * allowed, char ** words, int wordsSize) {
    int count = 0;
    int allowedMask = 0;

   
    for (int i = 0; allowed[i] != '\0'; i++) {
        allowedMask |= (1 << (allowed[i] - 'a'));
    }

    for (int i = 0; i < wordsSize; i++) {
        char* word = words[i];
        bool isValid = true;

        for (int j = 0; word[j] != '\0'; j++) {
            if (!(allowedMask & (1 << (word[j] - 'a')))) {
                isValid = false;
                break;
            }
        }

        if (isValid) {
            count++;
        }
    }

    return count;
}
