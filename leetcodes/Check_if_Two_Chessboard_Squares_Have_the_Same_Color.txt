
bool checkTwoChessboards(char* c1, char* c2) {
    int col1 = (c1[0] - 'a' + c1[1] - '1') % 2;
    int col2 = (c2[0] - 'a' + c2[1] - '1') % 2;
    return col1 == col2;
}
