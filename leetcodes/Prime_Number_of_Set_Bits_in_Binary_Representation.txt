#include <stdio.h>
#include <stdbool.h>


bool isPrime(int n) {
   return (1 << n) & (-1601558356);

}
int countSetBits(int n) {
   int count = 0;
   //brian kernigans algo always removes the last set bit/////
    while (n) {
     n &= (n - 1);  
         count++;
    }
    return count;
}

int countPrimeSetBits(int left, int right) {
    int count = 0;
    for (int i = left; i <= right; i++) {
        int setBits = countSetBits(i);
        if (isPrime(setBits)) {
            count++;
        }
    }
    return count;
}
