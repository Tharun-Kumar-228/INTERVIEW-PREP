int thirdMax(int* nums, int numsSize) {
    
    long first = LONG_MIN;
    long second = LONG_MIN;
    long third = LONG_MIN;

    for (int i = 0; i < numsSize; i++) {
        int n = nums[i];
        if (n == first || n == second || n == third)
            continue;

        if (n > first) {
            third = second;
            second = first;
            first = n;
        } else if (n > second) {
            third = second;
            second = n;
        } else if (n > third) {
            third = n;
        }
    }

    return third == LONG_MIN ? first : (int)third;
}
