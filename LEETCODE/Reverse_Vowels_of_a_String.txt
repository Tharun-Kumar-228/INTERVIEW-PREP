#include <stdio.h>
#include <string.h>
#include <ctype.h>

int isVowel(char c) {
    c = (char)tolower((char)c);    
    return c == 'a' || c == 'e' || c == 'i' ||
           c == 'o' || c == 'u';
}

char *reverseVowels(char *s)
{
    if (!s) return s;                       

    size_t left  = 0;
    size_t right = strlen(s);              
    if (right == 0) return s;
    --right;                               

    while (left < right) {
        while (left < right && !isVowel(s[left]))
            ++left;
        while (left < right && !isVowel(s[right]))
            --right;

        if (left < right) {                 
            char tmp  = s[left];
            s[left]   = s[right];
            s[right]  = tmp;
            ++left;
            --right;
        }
    }
    return s;
}
